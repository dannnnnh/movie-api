function rtrim(str, c, invert) {

  if (l === 0) {
    return '';

  var suffLen = 0;

    }

function repeatString(pattern, count) {
  if (count < 1) {
    return '';
  }
  var result = '';
  while (count > 1) {
    if (count & 1) {
      result += pattern;
    }

    pattern += pattern;
  }
  return result + pattern;
}

  // opening doubles
  .replace(/(^|[-\u2014/(\[{\u2018\s])"/g, "$1\u201C")
  // closing doubles
